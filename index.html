Projet de Synthèse : Implémentation et Optimisation de l'Algorithme de Dijkstra

Aperçu du projet

Ce projet académique visait à implémenter l'algorithme de Dijkstra pour la recherche du plus court chemin dans un graphe en utilisant trois structures de données différentes, puis à analyser et comparer leurs performances. Une interface graphique JavaFX a également été développée pour visualiser les graphes et leurs solutions.

Objectifs du projet

  • Implémenter l'algorithme de Dijkstra sur trois structures de données (tableau dynamique, arbre binaire complet, liste chaînée)
  • Analyser la complexité théorique et mesurer les performances pratiques des différentes implémentations
  • Développer une interface graphique interactive pour la visualisation des graphes
  • Créer des scripts d'automatisation pour générer et tester différents types de graphes
  • Mettre en pratique les concepts d'algorithmique, de structures de données et d'architecture logicielle

Implémentation de l'algorithme

L'algorithme de Dijkstra a été implémenté en C avec une approche générique permettant de l'utiliser avec différentes structures de données :

void Dijkstra(graph G, const char * source_name, int heap_type) {
    struct heap_t *H = new_heap(heap_type);
    struct vertex_t * v;
    for(int i=0; iheap_insert(H, get_vertex_total_distance(v),v);
    }

    while(!H->heap_is_empty(H)){
        v=get_heap_node_data(H->heap_extract_min(H));

        struct list_t* neighbours = get_vertex_incidence_list(v);
        struct list_node_t* iterr = get_list_head(neighbours);
        for(int i = 0; i< get_list_size(neighbours); i++){
            struct edge_t * n = get_list_node_data(iterr);
            struct vertex_t * V = get_edge_endpoint_V(n);
            struct vertex_t * U = get_edge_endpoint_U(n);
            if(strcmp(get_vertex_id(v),get_vertex_id(U)) == 0){ 
                U = V;
                V = v;
            }
            if(get_vertex_total_distance(U) > (get_vertex_total_distance(V) + get_edge_distance(n))){
                set_vertex_total_distance(U,(get_vertex_total_distance(V) + get_edge_distance(n)));
                set_vertex_predecessor(U,V);
                H->heap_increase_priority(H, get_vertex_dict_position(U), get_vertex_total_distance(U));
            }
            iterr = get_successor(iterr);
        }
    }
}

Design Patterns et Principes OO

Pattern Singleton

La classe Partie utilise le pattern Singleton pour garantir qu'une seule instance de partie existe:

public class Partie {
    private static Partie instance;
    
    private Partie() {
        // Initialisation
    }
    
    public static Partie getInstance() {
        if (instance == null) {
            instance = new Partie();
        }
        return instance;
    }
}

Défis techniques et solutions

Problème des tables en mutation

L'un des défis majeurs du projet était de gérer les "tables en mutation", qui se produisent lorsqu'un trigger tente de consulter une table déjà en cours de modification :

CREATE OR REPLACE TRIGGER CheckTotalQuotiteEmploye
BEFORE INSERT ON Participer
FOR EACH ROW
DECLARE
    jour DATE := :NEW.DateDebut;
    total_quotite NUMBER;
BEGIN
    WHILE jour <= :NEW.DateFin LOOP
        -- Utilisation d'une sous-requête WITH pour éviter l'erreur de mutation
        WITH TotalQuotite AS (
            SELECT SUM(Quotite) AS somme_quotite
            FROM Participer
            WHERE Matricule = :NEW.Matricule
              AND jour BETWEEN DateDebut AND DateFin
        )
        SELECT COALESCE(somme_quotite, 0) INTO total_quotite FROM TotalQuotite;

        -- Vérification du dépassement de quotité
        IF (total_quotite + :NEW.Quotite) > 100 THEN
            RAISE_APPLICATION_ERROR(-20009,
                'Le total des quotités de l''employé dépasse 100 pour la journée du '
                || TO_CHAR(jour, 'DD/MM/YYYY'));
        END IF;

        jour := jour + 1;
    END LOOP;
END;

Calcul des coûts par période

Le calcul des coûts mensuels des employés a nécessité une logique complexe pour gérer les périodes à cheval sur plusieurs mois :

CREATE OR REPLACE TRIGGER UpdateCoutMensuelEmploye
AFTER INSERT OR UPDATE OR DELETE ON Participer
FOR EACH ROW
DECLARE
    v_debut DATE;
    v_fin DATE;
    v_matricule NUMBER;
    v_cout_jour NUMBER;
    v_quotite NUMBER;
    v_mois DATE;
    v_nbj NUMBER;
    v_m NUMBER := 0;
BEGIN
    -- Déterminer la période à traiter
    CASE
        WHEN INSERTING THEN
            v_matricule := :NEW.Matricule;
            v_debut := :NEW.DateDebut;
            v_fin := :NEW.DateFin;
            v_quotite := :NEW.Quotite;
        WHEN DELETING THEN
            v_matricule := :OLD.Matricule;
            v_debut := :OLD.DateDebut;
            v_fin := :OLD.DateFin;
            v_quotite := :OLD.Quotite;
        WHEN UPDATING THEN
            -- Supprimer les anciennes entrées pour l'ancienne période
            DELETE FROM CoutMensuelEmploye
            WHERE Matricule = :OLD.Matricule
            AND TO_DATE(Mois, 'mm/yyyy') BETWEEN TRUNC(:OLD.DateDebut, 'MM') AND LAST_DAY(:OLD.DateFin);
            
            v_matricule := :NEW.Matricule;
            v_debut := :NEW.DateDebut;
            v_fin := :NEW.DateFin;
            v_quotite := :NEW.Quotite;
    END CASE;

    -- Récupérer le coût journalier de l'employé
    SELECT CoutJour INTO v_cout_jour
    FROM Employe
    WHERE Matricule = v_matricule;

    -- Si c'est une suppression, on supprime les entrées correspondantes
    IF DELETING THEN
        DELETE FROM CoutMensuelEmploye
        WHERE Matricule = v_matricule
        AND TO_DATE(Mois, 'mm/yyyy') BETWEEN TRUNC(v_debut, 'MM') AND LAST_DAY(v_fin);
    ELSE
        -- Pour INSERT et UPDATE, calculer les nouveaux coûts mensuels
        WHILE LAST_DAY(ADD_MONTHS(v_debut, v_m)) <= LAST_DAY(v_fin) LOOP
            v_mois := LAST_DAY(ADD_MONTHS(v_debut, v_m));
            -- Calculer le nombre de jours pour ce mois
            v_nbj := LEAST(v_fin, v_mois) - GREATEST(v_debut, TRUNC(v_mois, 'MM')) + 1;

            -- Mettre à jour ou insérer le coût mensuel
            MERGE INTO CoutMensuelEmploye c
            USING (
                SELECT
                    v_matricule as Matricule,
                    TO_CHAR(v_mois, 'mm/yyyy') as Mois,
                    v_cout_jour * v_nbj * (v_quotite/100) as NouveauCout
                FROM dual
            ) n
            ON (c.Matricule = n.Matricule AND c.Mois = n.Mois)
            WHEN MATCHED THEN
                UPDATE SET c.CoutMensuel = c.CoutMensuel + n.NouveauCout
            WHEN NOT MATCHED THEN
                INSERT (Matricule, Mois, CoutMensuel)
                VALUES (n.Matricule, n.Mois, n.NouveauCout);

            v_m := v_m + 1;
        END LOOP;
    END IF;
END;

Système d'alerte budgétaire

Un système d'alerte a été implémenté pour signaler les dépassements de budget :

CREATE OR REPLACE TRIGGER CheckBudgetParticiper
BEFORE INSERT OR UPDATE ON Participer
FOR EACH ROW
DECLARE
    budget NUMBER;
    cout_emp_actuel NUMBER;
    cout_mat_actuel NUMBER;
    cout_ajoute NUMBER;
    mat_resp NUMBER;
    nom_resp VARCHAR2(100);
BEGIN
    -- Récupérer les données du projet
    SELECT Budget, CoutEmp, CoutMat, RespProj
    INTO budget, cout_emp_actuel, cout_mat_actuel, mat_resp
    FROM Projet
    WHERE IdProj = :NEW.IdProj;

    -- Calculer le coût ajouté
    SELECT e.CoutJour * (:NEW.DateFin - :NEW.DateDebut + 1) * :NEW.Quotite / 100
    INTO cout_ajoute
    FROM Employe e
    WHERE e.Matricule = :NEW.Matricule;

    -- Vérifier le dépassement de budget
    IF (cout_emp_actuel + cout_mat_actuel + cout_ajoute) > budget THEN
        -- Récupérer le nom du responsable
        SELECT Nom INTO nom_resp
        FROM Employe
        WHERE Matricule = mat_resp;

        -- Afficher l'alerte
        DBMS_OUTPUT.PUT_LINE('Alerte : Budget dépassé sur le projet ' || :NEW.IdProj ||
                             ' dirigé par ' || nom_resp ||
                             '. Coût en matériel : ' || cout_mat_actuel ||
                             ', coût employé : ' || cout_emp_actuel ||
                             ', budget : ' || budget);
    END IF;
END;

  • Message adapté si aucun véhicule n'est disponible dans le type demandé

Fonction FormuleAvantageuse

FormuleAvantageuse(

Pattern Visitor

Le pattern Visitor est utilisé pour séparer les algorithmes des structures d'objets. Dans ce projet, il est implémenté pour les opérations de dessin et de sauvegarde :

// FormeVisitor.h - Interface du visiteur
class FormeVisitor {
public:
    virtual ~FormeVisitor() {}
    virtual void visit(const Cercle & c) = 0;
    virtual void visit(const Segment & s) = 0;
    virtual void visit(const Triangle & t) = 0;
    virtual void visit(const Polygone & p) = 0;
    virtual void visit(const Groupe & g) = 0;
};

// ClientDessinVisitor.h - Visiteur pour générer des commandes de dessin
class ClientDessinVisitor : public FormeVisitor {
private:
    ostringstream commandes;
public:
    void visit(const Cercle &c) override {
        commandes << "draw Cercle{ " << c.getCentre() << ", " << c.getRayon() << ", " << c.getCouleur() << " }\n";
    }
    
    // Autres méthodes visit pour les différentes formes...
    
    string getCommandes() const {
        return commandes.str();
    }
};

Pattern Chain of Responsibility

Le pattern Chain of Responsibility est utilisé pour les parseurs de formes, permettant l'extension facile avec de nouvelles formes :

// ParseurFormeCor.h - Base de la chaîne de responsabilité
class ParseurFormeCor {
protected:
    const ParseurFormeCor* suivant;
public:
    ParseurFormeCor(const ParseurFormeCor* suivant = nullptr) : suivant(suivant) {}
    virtual ~ParseurFormeCor() { delete suivant; }
    
    virtual bool parse(std::istream &in, std::string &line, Forme*& r) const {
        if (suivant) {
            return suivant->parse(in, line, r);
        }
        return false;
    }
};

// ParseurFormeCorExpert.h - Organisation de la chaîne
ParseurFormeCorExpert() : ParseurFormeCor(
    new ParseurGroupe(

                )
            )
        )
    )
) {}

Pattern Singleton

Le pattern Singleton est utilisé pour la gestion de l'initialisation Winsock :

// WinsockManager.h - Implémentation du Singleton
class WinsockManager {
private:
    // Constructeur privé
    WinsockManager() {
        WSADATA wsaData;
        int r = WSAStartup(MAKEWORD(2, 0), &wsaData);
        if (r != 0) {
            std::ostringstream oss;
            oss << "WSAStartup a échoué, code erreur : " << r;
            throw std::runtime_error(oss.str());
        }
    }
    // Destructeur privé
    ~WinsockManager() {
        WSACleanup();
    }
public:
    // Supprimer le constructeur de copie et l'opérateur d'affectation
    WinsockManager(const WinsockManager&) = delete;
    WinsockManager& operator=(const WinsockManager&) = delete;

    // Méthode d'accès statique au singleton
    static WinsockManager& instance() {
        static WinsockManager instance;
        return instance;
    }
};
